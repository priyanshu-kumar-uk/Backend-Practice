 const user = await userModel.aggregate([
        {
            $search: {
                index: 'user_search_feature',
                autocomplete: {
                    query: queryData,  /*Search user in cuurent collection(userModel)*/
                    path: 'username',
                }
            }
        },   
        {
            $lookup: {
                from: "follows",   
                as: "followData",
                let: { searchUser: "$username" },  /* Only we want search users */
                pipeline: [
                    {
                        $match: {
                            $expr: {
                                $and: [
                                    { $eq: ["$follower", id] },    /*Login user*/
                                    { $eq: ["$followee", "$$searchUser"] }
                                ]
                            }
                        }
                    }
                ]
            }
        },
        {
            $addFields: {
                followstatus: {        /* add field follow data*/
                    $cond: {
                        if: { $lt: [{ $size: "$followData" }, 1] },
                        then: null,
                        else: {
                            $cond: {
                                if: {
                                    $eq: [
                                        { $arrayElemAt: ["$followData.status", 0] },
                                        "prnding"
                                    ]
                                },
                                then: "requested",
                                else: "following"
                            }
                        }
                    }
                }
            }
        },
        {
            $project: {
                username: 1,
                fullname: 1,
                profilePicture: 1,
                followstatus: 1,    /*Jo data chaiye ya nahi chaiye apk output mai*/
            }
        }
    ])
